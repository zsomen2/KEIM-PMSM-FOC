8bool MUCI_ReceivePacket(MUCI_Packet_t* Message, uint8_t DataBuffer[], uint32_t FIFODataLength)



9{



10    uint32_t FIFODataPointer;



11    uint8_t ReceivedData;



12



13    for(FIFODataPointer = 0u; FIFODataPointer < FIFODataLength; FIFODataPointer++)



14    {



15        ReceivedData =  DataBuffer[FIFODataPointer];



16        if (MUCI_DLE == ReceivedData)



17        {



18            Message->DLEWas = true;



19            continue;



20        }



21        if (true == Message->DLEWas)



22        {



23            ReceivedData= ReceivedData ^ MUCI_XDT;



24            Message->DLEWas = false;



25        }



26



27        switch (Message->State)



28        {



29            case MUCI_PACKET_IDLE:



30                break;



31            case MUCI_PACKET_RECEIVE_WAIT_FOR_STX:



32                if (MUCI_STX == ReceivedData)



33                {



34                    Message->CRC = 0xFFFFu;



35                    Message->DataPointer = 0u;



36                    Message->State = MUCI_PACKET_RECEIVE_ADDRESS;



37                }



38                break;



39            case MUCI_PACKET_RECEIVE_ADDRESS:



40                Message->Address = ReceivedData;



41                MUCI_CalcCheckSum(&Message->CRC, ReceivedData);



42                Message->State = MUCI_PACKET_RECEIVE_COMMAND;



43                break;



44            case MUCI_PACKET_RECEIVE_COMMAND:



45                Message->Command = (MUCI_Commands)ReceivedData;



46                MUCI_CalcCheckSum(&Message->CRC, ReceivedData);



47                Message->State = MUCI_PACKET_RECEIVE_KEY;



48                break;



49            case MUCI_PACKET_RECEIVE_KEY:



50                Message->Key = ReceivedData;



51                MUCI_CalcCheckSum(&Message->CRC, ReceivedData);



52                Message->State = MUCI_PACKET_RECEIVE_LENGTH_HIGHBYTE;



53                break;



54            case MUCI_PACKET_RECEIVE_LENGTH_HIGHBYTE:



55                Message->DataLength = ((uint16_t)ReceivedData) << 8;



56                MUCI_CalcCheckSum(&Message->CRC, ReceivedData);



57                Message->State = MUCI_PACKET_RECEIVE_LENGTH_LOWBYTE;



58                break;



59            case MUCI_PACKET_RECEIVE_LENGTH_LOWBYTE:



60                Message->DataLength = Message->DataLength | (uint16_t)ReceivedData;



61                MUCI_CalcCheckSum(&Message->CRC, ReceivedData);



62                if (MUCI_DATA_LENGTH < Message->DataLength)



63                {



64                    /* Not enough buffer for the message */



65                    MUCI_Counters.NotEnoughSpaceInTheBuffer++;



66                    Message->State = MUCI_PACKET_RECEIVE_WAIT_FOR_STX;



67                }



68                else if (0u == Message->DataLength)



69                {



70                    /* No data in the package */



71                    Message->State = MUCI_PACKET_RECEIVE_CRC_LOWBYTE;



72                }



73                else



74                {



75                    Message->State = MUCI_PACKET_RECEIVE_DATA;



76                }



77                break;



78            case MUCI_PACKET_RECEIVE_DATA:



79                Message->Data[Message->DataPointer] = ReceivedData;



80                Message->DataPointer = Message->DataPointer + 1u;



81                MUCI_CalcCheckSum(&Message->CRC, ReceivedData);



82                if (Message->DataPointer >= Message->DataLength)



83                {



84                    Message->State = MUCI_PACKET_RECEIVE_CRC_LOWBYTE;



85                }



86                break;



87            case MUCI_PACKET_RECEIVE_CRC_LOWBYTE:



88                MUCI_CalcCheckSum(&Message->CRC, ReceivedData);



89                Message->State = MUCI_PACKET_RECEIVE_CRC_HIGHBYTE;



90                break;



91            case MUCI_PACKET_RECEIVE_CRC_HIGHBYTE:



92                MUCI_CalcCheckSum(&Message->CRC, ReceivedData);



93                if (0u == Message->CRC)



94                {



95                    Message->State = MUCI_PACKET_RECEIVE_ETX;



96                }



97                else



98                {



99                    MUCI_Counters.WrongCRC++;



100                    Message->State = MUCI_PACKET_RECEIVE_WAIT_FOR_STX;



101                }



102                break;



103            case MUCI_PACKET_RECEIVE_ETX:



104                if (MUCI_ETX == ReceivedData)



105                {



106                    /* Full package received with good CRC */



107                    MUCI_Counters.GoodPacketReceived++;



108                    Message->State = MUCI_PACKET_RECEIVE_FINISHED;



109                }



110                else



111                {



112                    MUCI_Counters.NoETXCharacterReceived++;



113                    Message->State = MUCI_PACKET_RECEIVE_WAIT_FOR_STX;



114                }



115                break;



116            case MUCI_PACKET_RECEIVE_FINISHED:



117                break;



118            default:



119                MUCI_Counters.WTF++;



120                Message->State = MUCI_PACKET_IDLE;



121                break;



122        }



123    }



124    return true;



125}

127bool MUCI_ProcessPacket(MUCI_Packet_t* Message, MUCI_Packet_t* Answer)



128{



129    if ((MUCI_PACKET_RECEIVE_FINISHED == Message->State) && (MUCI_PACKET_SEND_FINISHED == Answer->State))



130    {



131        if (MUCI_GetDeviceAddress() == Message->Address)



132        {



133            /* Address is own */



134            switch(Message->Command)



135            {



136            	case MUCI_GET_PROTOCOL_VERSION:



137            		MUCI_Command_GET_PROTOCOL_VERSION(Message, Answer);



138            		break;



139                case MUCI_PING:



140                    MUCI_Command_PING(Message, Answer);



141                    break;



142                case MUCI_READ:



143                    MUCI_Command_READ(Message, Answer);



144                    break;



145                case MUCI_WRITE:



146                    MUCI_Command_WRITE(Message, Answer);



147                    break;



148                case MUCI_READ_NEW:



149                    MUCI_Command_READ_NEW(Message, Answer);



150                    break;



151                case MUCI_WRITE_NEW:



152                    MUCI_Command_WRITE_NEW(Message, Answer);



153                    break;



154                case MUCI_READ_TABNAMES:



155                    MUCI_Command_READ_TABNAMES(Message, Answer);



156                    break;



157                case MUCI_READ_MON_TAB0:



158                case MUCI_READ_MON_TAB1:



159                case MUCI_READ_MON_TAB2:



160                case MUCI_READ_MON_TAB3:



161                case MUCI_READ_MON_TAB4:



162                case MUCI_READ_MON_TAB5:



163                case MUCI_READ_MON_TAB6:



164                case MUCI_READ_MON_TAB7:



165                case MUCI_READ_MON_TAB8:



166                case MUCI_READ_MON_TAB9:



167                    MUCI_Command_READ_MON_TABS(Message, Answer);



168                    break;



169                case MUCI_READ_UNITS:



170                    MUCI_Command_READ_UNITS(Message, Answer);



171                    break;



172                case MUCI_READ_TAB0VALUES:



173                case MUCI_READ_TAB1VALUES:



174                case MUCI_READ_TAB2VALUES:



175                case MUCI_READ_TAB3VALUES:



176                case MUCI_READ_TAB4VALUES:



177                case MUCI_READ_TAB5VALUES:



178                case MUCI_READ_TAB6VALUES:



179                case MUCI_READ_TAB7VALUES:



180                case MUCI_READ_TAB8VALUES:



181                case MUCI_READ_TAB9VALUES:



182                    MUCI_Command_READ_TABVALUES(Message, Answer);



183                    break;



184                case MUCI_REBOOT:



185                    MUCI_Command_REBOOT(Message, Answer);



186                    break;



187                case MUCI_BOOT_SEND_API:



188                    MUCI_Command_BOOT_SEND_API(Message, Answer);



189                    break;



190                case MUCI_READ_BBX:



191                    MUCI_Command_MUCI_READ_BBX(Message, Answer);



192                    break;



193                case MUCI_BOOT_CLEAR_SECTOR:



194                    MUCI_Command_BOOT_CLEAR_SECTOR(Message, Answer);



195                    break;



196                case MUCI_BOOT_READ_FLASH:



197                	MUCI_Command_BOOT_READ_FLASH(Message, Answer);



198                	break;



199                case MUCI_BOOT_PROGRAM_FLASH:



200                    MUCI_Command_BOOT_PROGRAM_FLASH(Message, Answer);



201                    break;



202                default:



203                    MUCI_Counters.NotSupportedCommand++;



204                    Message->State = MUCI_PACKET_RECEIVE_WAIT_FOR_STX;



205                    break;



206            }



207        }



208        else



209        {



210            /* Skip packet not own address */



211            //Message->State = MUCI_PACKET_RECEIVE_WAIT_FOR_STX;



212        }



213



214    }



215    return true;



216}

218bool MUCI_PacketRouterSend(MUCI_Packet_t* Source, MUCI_Packet_t* Destination, uint8_t MyAddress)



219{



220    if ((MUCI_PACKET_RECEIVE_FINISHED == Source->State) && (MUCI_PACKET_RECEIVE_WAIT_FOR_STX == Destination->State))



221    {



222        if (MyAddress == Source->Address)



223        {



224            memcpy(Destination, Source, sizeof(MUCI_Packet_t));



225            Source->State = MUCI_PACKET_RECEIVE_WAIT_FOR_STX;



226            Destination->State = MUCI_PACKET_RECEIVE_FINISHED;



227            return true;



228        }



229        else



230        {



231            return false;



232        }



233    }



234    else



235    {



236        return false;



237    }



238}

271bool MUCI_SendPacket(MUCI_Packet_t* Message, uint8_t DataBuffer[], uint32_t* DataLength)



272{



273    uint32_t DataPointer;



274    uint8_t SendData;



275



276    for (DataPointer = 0; DataPointer < *DataLength; DataPointer++)



277    {



278        switch(Message->State)



279        {



280            case MUCI_PACKET_IDLE:



281                break;



282            case MUCI_PACKET_SEND_STX:



283                SendData = MUCI_STX;



284                Message->CRC = 0xFFFFu;



285                Message->DataPointer = 0u;



286                Message->NextState = MUCI_PACKET_SEND_ADDRESS;



287                break;



288            case MUCI_PACKET_SEND_ADDRESS:



289                SendData = Message->Address;



290                Message->NextState = MUCI_PACKET_SEND_COMMAND;



291                break;



292            case MUCI_PACKET_SEND_COMMAND:



293                SendData = Message->Command;



294                Message->NextState = MUCI_PACKET_SEND_KEY;



295                break;



296            case MUCI_PACKET_SEND_KEY:



297                SendData = Message->Key;



298                Message->NextState = MUCI_PACKET_SEND_LENGTH_HIGHBYTE;



299                break;



300            case MUCI_PACKET_SEND_LENGTH_HIGHBYTE:



301                SendData = (Message->DataLength >> 8u) & 0xFFu;



302                Message->NextState = MUCI_PACKET_SEND_LENGTH_LOWBYTE;



303                break;



304            case MUCI_PACKET_SEND_LENGTH_LOWBYTE:



305                SendData = Message->DataLength & 0xFFu;



306                if (0u == Message->DataLength)



307                {



308                    /* No data in the command */



309                    Message->NextState = MUCI_PACKET_SEND_CRC_LOWBYTE;



310                }



311                else



312                {



313                    Message->NextState = MUCI_PACKET_SEND_DATA;



314                }



315                break;



316            case MUCI_PACKET_SEND_DATA:



317                SendData = Message->Data[Message->DataPointer];



318                Message->NextState = MUCI_PACKET_SEND_DATA;



319                break;



320            case MUCI_PACKET_SEND_CRC_LOWBYTE:



321                SendData = Message->CRC & 0xFFu;



322                Message->NextState = MUCI_PACKET_SEND_CRC_HIGHBYTE;



323                break;



324            case MUCI_PACKET_SEND_CRC_HIGHBYTE:



325                SendData = (Message->CRC >> 8u) & 0xFFu;



326                Message->NextState = MUCI_PACKET_SEND_ETX;



327                break;



328            case MUCI_PACKET_SEND_ETX:



329                SendData = MUCI_ETX;



330                Message->NextState = MUCI_PACKET_SEND_FINISHED;



331                break;



332            case MUCI_PACKET_SEND_FINISHED:



333                break;



334            default:



335                MUCI_Counters.WTF++;



336                Message->State = MUCI_PACKET_IDLE;



337                break;



338        }



339



340        if ((MUCI_PACKET_SEND_STX == Message->State) || (MUCI_PACKET_SEND_ETX == Message->State))



341        {



342            DataBuffer[DataPointer] = SendData;



343            Message->State = Message->NextState;



344        }



345        else if ((MUCI_PACKET_IDLE == Message->State) || (MUCI_PACKET_SEND_FINISHED == Message->State))



346        {



347            *DataLength = DataPointer;



348            break;



349        }



350        else



351        {



352            if ((true != Message->DLEWas) && MUCI_NeedDataDLE(SendData))



353            {



354                DataBuffer[DataPointer] = MUCI_DLE;



355                Message->DLEWas = true;



356            }



357            else if (MUCI_NeedDataDLE(SendData))



358            {



359                DataBuffer[DataPointer] =SendData ^ MUCI_XDT;



360                Message->DLEWas = false;



361                if ((MUCI_PACKET_SEND_CRC_LOWBYTE != Message->State) && (MUCI_PACKET_SEND_CRC_HIGHBYTE != Message->State))



362                {



363                    MUCI_CalcCheckSum(&Message->CRC, SendData & 0xFFu);



364                }



365                if (MUCI_PACKET_SEND_DATA == Message->State)



366                {



367                    Message->DataPointer++;



368                    if (Message->DataPointer >= Message->DataLength)



369                    {



370                        Message->NextState = MUCI_PACKET_SEND_CRC_LOWBYTE;



371                    }



372                }



373                Message->State = Message->NextState;



374            }



375            else



376            {



377                DataBuffer[DataPointer] = SendData;



378                if ((MUCI_PACKET_SEND_CRC_LOWBYTE != Message->State) && (MUCI_PACKET_SEND_CRC_HIGHBYTE != Message->State))



379                {



380                    MUCI_CalcCheckSum(&Message->CRC, SendData & 0xFFu);



381                }



382                if (MUCI_PACKET_SEND_DATA == Message->State)



383                {



384                    Message->DataPointer++;



385                    if (Message->DataPointer >= Message->DataLength)



386                    {



387                        Message->NextState = MUCI_PACKET_SEND_CRC_LOWBYTE;



388                    }



389                }



390                Message->State = Message->NextState;



391            }



392        }



393    }



394    return true;



395}

397bool MUCI_NeedDataDLE(uint8_t data)



398{



399    bool ReturnValue = false;



400



401    switch (data & 0xFFu)



402    {



403      case MUCI_STX:



404      case MUCI_ETX:



405      case MUCI_DLE:



406          ReturnValue = true;



407       break;



408      default:



409          ReturnValue = false;



410       break;



411    }



412    return ReturnValue;
