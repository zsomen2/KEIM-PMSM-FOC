54void MUCI_CalcCheckSum(uint16_t *chksum, uint8_t data)



55{



56  uint8_t uchCRCHi = *chksum >> 8u;   /* high byte of CRC initialized   */



57  uint8_t uchCRCLo = *chksum & 0xFFu; /* low byte of CRC initialized   */



58  uint8_t uIndex; /* will index into CRC lookup table   */



59



60  uIndex = uchCRCLo ^ data; /* calculate the CRC   */



61  uchCRCLo = uchCRCHi ^ MUCI_CRCGenerator_auchCRCHi[uIndex];



62  uchCRCHi = MUCI_CRCGenerator_auchCRCLo[uIndex];



63



64  *chksum = (uchCRCHi << 8 | uchCRCLo);



65}
