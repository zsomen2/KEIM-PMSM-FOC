47void SPI_Init(SPIStruct_t *SPI)



48{



49  SPI_Init_Clocks();



50



51  SPI->SPIInst->SPICCR.bit.SPISWRESET = 0;      // 7      SPI SW Reset



52  SPI->SPIInst->SPICCR.bit.CLKPOLARITY = 1;     // 6      Clock polarity=Falling edge



53  SPI->SPIInst->SPICCR.bit.SPILBK = 0;          // 4      Loop-back disable



54  SPI->SPIInst->SPICCR.bit.SPICHAR = 15;        // 3:0    Character length control = 16



55  SPI->SPIInst->SPICTL.bit.OVERRUNINTENA = 0;   // 4      Overrun interrupt disable



56  SPI->SPIInst->SPICTL.bit.CLK_PHASE = 1;       // 3      Clock phase select = 1, Delayed



57  SPI->SPIInst->SPICTL.bit.MASTER_SLAVE = 1;    // 2      Network control mode = master



58  SPI->SPIInst->SPICTL.bit.TALK = 1;            // 1      Master/Slave transmit enable



59  SPI->SPIInst->SPICTL.bit.SPIINTENA = 0;       // 0      Interrupt disable



60  SPI->SPIInst->SPIBRR = LSPCLK/3000000-1;      //        Baudrate = 10MHz / (9 + 1) = 3MHz



61  SPI->SPIInst->SPICCR.bit.SPISWRESET = 1;      // 7      SPI SW Reset: not active



62



63  SPI->SPIInst->SPIFFTX.bit.SPIRST = 0;         // 15     Reset SPI



64  SPI->SPIInst->SPIFFTX.bit.SPIFFENA = 1;       // 14     Enhancement enable



65  SPI->SPIInst->SPIFFTX.bit.TXFIFO = 0;         // 13     FIFO reset



66



67  SPI->SPIInst->SPIFFRX.bit.RXFFOVFCLR = 1;     // 14     Clear overflow



68  SPI->SPIInst->SPIFFRX.bit.RXFIFORESET = 0;    // 13     FIFO reset



69



70  SPI->SPIInst->SPIFFTX.bit.SPIRST = 1;         // 15     enable SPI



71  SPI->SPIInst->SPIFFTX.bit.TXFIFO = 1;         // 13     FIFO normal op



72



73  SPI->SPIInst->SPIFFRX.bit.RXFIFORESET = 1;    // 13     FIFO normal op



74



75  SPI->SPIInst->SPIFFCT.bit.TXDLY = 2;          // 7:0    FIFO transmit delay, 2401A ennyit kesleltet



76



77  SPI_Init_GPIOs();



78}

80void SPI_CS_Assert(SPICSState_t State)



81{



82  switch(State)



83  {



84    case HIGH:



85      GpioDataRegs.GPASET.bit.GPIO15 = 1;



86      break;



87    case LOW:



88      GpioDataRegs.GPACLEAR.bit.GPIO15 = 1;



89      break;



90    default:



91      GpioDataRegs.GPASET.bit.GPIO15 = 1;



92      break;



93  }



94}

96void SPI_Set_CLKMode(SPIStruct_t *SPI, SPICLKMode_t CLKMode)



97{



98



99  SPI->Mode = CLKMode;



100



101  switch (SPI->Mode)



102  {



103    case SPI_MODE_A:



104      SPI->SPIInst->SPICCR.bit.CLKPOLARITY = 0;



105	  SPI->SPIInst->SPICTL.bit.CLK_PHASE = 0;



106      break;



107    case SPI_MODE_B:



108      SPI->SPIInst->SPICCR.bit.CLKPOLARITY = 0;



109  	  SPI->SPIInst->SPICTL.bit.CLK_PHASE = 1;



110      break;



111    case SPI_MODE_C:



112      SPI->SPIInst->SPICCR.bit.CLKPOLARITY = 1;



113  	  SPI->SPIInst->SPICTL.bit.CLK_PHASE = 0;



114      break;



115    case SPI_MODE_D:



116      SPI->SPIInst->SPICCR.bit.CLKPOLARITY = 1;



117  	  SPI->SPIInst->SPICTL.bit.CLK_PHASE = 1;



118      break;



119    default:



120      SPI->SPIInst->SPICCR.bit.CLKPOLARITY = 0;



121  	  SPI->SPIInst->SPICTL.bit.CLK_PHASE = 0;



122      break;



123  }



124}

126void SPI_Set_DataLength(SPIStruct_t *SPI, SPIDataLength_t DataLength)



127{



128



129  SPI->DataLength = DataLength;



130



131  switch (SPI->DataLength)



132  {



133    case DATALENGTH_8BIT:



134      SpiaRegs.SPICCR.bit.SPICHAR = 7;        //Character length control = 8



135      break;



136    case DATALENGTH_16BIT:



137      SpiaRegs.SPICCR.bit.SPICHAR = 15;        //Character length control = 16



138      break;



139    default:



140      SpiaRegs.SPICCR.bit.SPICHAR = 15;        //Character length control = 16



141      break;



142  }



143}

151void SPI_Transmit_8bit(SPIStruct_t *SPI, uint8_t Data)



152{



153  SPI_Set_DataLength(SPI, DATALENGTH_8BIT);



154



155  SPI->SPIInst->SPITXBUF = Data;



156  while (SPI->SPIInst->SPIFFTX.bit.TXFFST != 0);



157}

167void SPI_Transmit_32bit(SPIStruct_t *SPI, uint32_t Data)



168{



169  SPI_Set_DataLength(SPI, DATALENGTH_16BIT);



170



171  SPI->SPIInst->SPITXBUF = Data >> 16;



172  SPI->SPIInst->SPITXBUF = Data & 0xFFFF;



173  while (SPI->SPIInst->SPIFFTX.bit.TXFFST != 0);



174}
